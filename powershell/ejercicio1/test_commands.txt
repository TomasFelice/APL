Write-Host "=== INICIANDO LOTE DE PRUEBAS EJERCICIO 1 POWERSHELL (ACTUALIZADO) ==="

# =================================================
# 1. PRUEBAS DE AYUDA (Get-Help nativo)
# =================================================
Write-Host "=== 1. PRUEBAS DE AYUDA (Get-Help nativo) ==="

Write-Host "1.1 Ayuda básica con Get-Help"
Get-Help ./powershell/ejercicio1/ejercicio1.ps1

Write-Host "`n1.2 Ayuda con ejemplos"
Get-Help ./powershell/ejercicio1/ejercicio1.ps1 -Examples

Write-Host "`n1.3 Ayuda detallada"
Get-Help ./powershell/ejercicio1/ejercicio1.ps1 -Detailed

Write-Host "`n1.4 Ayuda completa"
Get-Help ./powershell/ejercicio1/ejercicio1.ps1 -Full

# =================================================
# 2. PRUEBAS DE PARÁMETROS OBLIGATORIOS Y ParameterSets
# =================================================
Write-Host "`n=== 2. PRUEBAS DE PARÁMETROS OBLIGATORIOS Y ParameterSets ==="

Write-Host "2.1 Sin parámetros - debe solicitar parámetros obligatorios"
Write-Host "NOTA: El script solicitará interactivamente los parámetros faltantes"
# Simular entrada vacía para evitar bloqueo:
echo "" | ./powershell/ejercicio1/ejercicio1.ps1

Write-Host "`n2.2 Solo directorio sin especificar salida"
Write-Host "NOTA: Debe fallar porque falta especificar -Archivo o -Pantalla"
echo "" | ./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_normal

Write-Host "`n2.3 Conflicto de ParameterSets - -Archivo y -Pantalla juntos"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_normal -Archivo salida.json -Pantalla

Write-Host "`n2.4 Directorio vacío (validación NotNullOrEmpty)"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio "" -Pantalla

Write-Host "`n2.5 ParameterSet ArchivoSet válido"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_archivo_unico -Archivo ./powershell/ejercicio1/out/test_parameterset.json

Write-Host "`n2.6 ParameterSet PantallaSet válido"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_archivo_unico -Pantalla

# =================================================
# 3. PRUEBAS DE DIRECTORIO INEXISTENTE/INVÁLIDO
# =================================================
Write-Host "`n=== 3. PRUEBAS DE DIRECTORIO INEXISTENTE/INVÁLIDO ==="

Write-Host "3.1 Directorio inexistente"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio /directorio/inexistente -Pantalla

Write-Host "`n3.2 Directorio sin archivos .txt"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_vacio -Pantalla

Write-Host "`n3.3 Archivo en lugar de directorio"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/ejercicio1.ps1 -Pantalla

# =================================================
# 4. PRUEBAS DE RUTAS RELATIVAS Y ABSOLUTAS
# =================================================
Write-Host "`n=== 4. PRUEBAS DE RUTAS RELATIVAS Y ABSOLUTAS ==="

Write-Host "4.1 Ruta relativa con ./"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_normal -Pantalla

Write-Host "`n4.2 Ruta relativa sin ./"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio powershell/ejercicio1/in/caso_normal -Pantalla

Write-Host "`n4.3 Ruta absoluta"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio /home/tfelice/dev/vh/APL/powershell/ejercicio1/in/caso_normal -Pantalla

# =================================================
# 5. PRUEBAS FUNCIONALES - CASOS NORMALES
# =================================================
Write-Host "`n=== 5. PRUEBAS FUNCIONALES - CASOS NORMALES ==="

Write-Host "5.1 Caso normal - múltiples archivos, salida a pantalla"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_normal -Pantalla

Write-Host "`n5.2 Caso normal - múltiples archivos, salida a archivo"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_normal -Archivo ./powershell/ejercicio1/out/caso_normal/resultado_completo.json

Write-Host "`n5.3 Verificar archivo generado"
Get-Content ./powershell/ejercicio1/out/caso_normal/resultado_completo.json

Write-Host "`n5.4 Prueba con -Verbose para ver archivos temporales"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_normal -Pantalla -Verbose

# =================================================
# 6. PRUEBAS DE ARCHIVOS TEMPORALES Y CLEANUP
# =================================================
Write-Host "`n=== 6. PRUEBAS DE ARCHIVOS TEMPORALES Y CLEANUP ==="

Write-Host "6.1 Verificar archivos temporales durante ejecución"
Write-Host "Contar archivos temp antes:"
find /tmp -name "*canal*" -o -name "*ejercicio1*" 2>/dev/null | wc -l
Write-Host "Ejecutar con -Verbose para ver archivos temporales:"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_normal -Pantalla -Verbose
Write-Host "Contar archivos temp después (deben eliminarse):"
sleep 1
find /tmp -name "*canal*" -o -name "*ejercicio1*" 2>/dev/null | wc -l

Write-Host "`n6.2 Verificar cleanup en caso de error"
Write-Host "NOTA: Los archivos temporales deben eliminarse incluso con errores"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_datos_invalidos -Pantalla -Verbose
sleep 1
find /tmp -name "*canal*" -o -name "*ejercicio1*" 2>/dev/null | wc -l

# =================================================
# 7. PRUEBAS CON ARCHIVO ÚNICO
# =================================================
Write-Host "`n=== 7. PRUEBAS CON ARCHIVO ÚNICO ==="

Write-Host "7.1 Archivo único - datos mínimos"

./powershell/ejercicio1/ejercicio1.ps1 -d ./powershell/ejercicio1/in/caso_archivo_unico -a ./powershell/ejercicio1/out/caso_archivo_unico/cambio_año.json

# 8. PRUEBAS CON DATOS INVÁLIDOS
# =================================================
Write-Host "`n=== 8. PRUEBAS CON DATOS INVÁLIDOS ==="

Write-Host "8.1 Archivos completamente vacíos"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_solo_vacios -Pantalla

Write-Host "`n8.2 Archivos con formato incorrecto"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_datos_invalidos -Pantalla

# 9. PRUEBAS DE ORDEN DE PARÁMETROS

# 10. PRUEBAS DE ARCHIVOS DE SALIDA CON ERRORES
# =================================================
Write-Host "`n=== 10. PRUEBAS DE ARCHIVOS DE SALIDA CON ERRORES ==="

Write-Host "10.1 Directorio de salida inexistente"

Write-Host "`n10.2 Archivo sin permisos de escritura (simulado con ruta inválida)"

# =================================================

./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in -Archivo ./powershell/ejercicio1/out/casos_especiales/datos_historicos.json

Get-Content ./powershell/ejercicio1/out/casos_especiales/datos_historicos.json

# 11. PRUEBAS DE VERIFICACIÓN DE TIPOS POWERSHELL
# =================================================
Write-Host "`n=== 11. PRUEBAS ESPECÍFICAS DE POWERSHELL ==="

Write-Host "11.1 Verificar que -Pantalla es un switch (no necesita valor)"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_archivo_unico -Pantalla

Write-Host "`n11.2 Verificar parámetros con nombres completos"
./powershell/ejercicio1/ejercicio1.ps1 -Directorio ./powershell/ejercicio1/in/caso_archivo_unico -Pantalla

Write-Host "`n11.3 Verificar parámetros con aliases"
./powershell/ejercicio1/ejercicio1.ps1 -d ./powershell/ejercicio1/in/caso_archivo_unico -p

Write-Host "12.1 Verificar que no quedan archivos temporales"
Get-ChildItem -Path $env:TEMP -Filter "tmp*" | Where-Object { $_.Name -like "*ejercicio1*" } | ForEach-Object { Write-Host "Archivo temporal encontrado: $($_.FullName)" }
